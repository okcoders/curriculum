Express Setup

const express = require("express");
const app = express();

// Middleware to parse JSON bodies
app.use(express.json());

// Start server
const PORT = 3000;
app.listen(PORT, () => {
  console.log("Server running on port " + PORT);
});
Important: app.use(express.json()) must come BEFORE your routes, or req.body will be undefined!

HTTP Methods

Basic Routes

let expenses = [];
let nextId = 1;

// GET all expenses
app.get("/api/expenses", (req, res) => {
  res.json(expenses);
});

// GET single expense
app.get("/api/expenses/:id", (req, res) => {
  const id = parseInt(req.params.id);
  const expense = expenses.find(e => e.id === id);
  
  if (!expense) {
    return res.status(404).json({ error: "Not found" });
  }
  res.json(expense);
});

// POST new expense
app.post("/api/expenses", (req, res) => {
  const { description, amount, category } = req.body;
  
  const expense = {
    id: nextId++,
    description,
    amount,
    category
  };
  
  expenses.push(expense);
  res.status(201).json(expense);
});

// DELETE expense
app.delete("/api/expenses/:id", (req, res) => {
  const id = parseInt(req.params.id);
  const index = expenses.findIndex(e => e.id === id);
  
  if (index === -1) {
    return res.status(404).json({ error: "Not found" });
  }
  
  expenses.splice(index, 1);
  res.status(204).send();
});

Request Object

app.get("/api/expenses/:id", (req, res) => {
  req.params.id    // URL parameter - from /api/expenses/123
  req.query.sort   // Query string - from ?sort=date
  req.body         // Request body - from POST/PUT JSON
});

Response Object

res.json({ data: "hello" });        // Send JSON
res.status(201).json(expense);      // Set status + JSON
res.status(404).json({ error: "Not found" });
res.status(204).send();             // No content

Route Order Matters

// WRONG - :id catches "summary"
app.get("/api/expenses/:id", handler);
app.get("/api/expenses/summary", handler); // Never reached!

// RIGHT - specific routes first
app.get("/api/expenses/summary", handler);
app.get("/api/expenses/:id", handler);

Common Mistakes

Mistake Problem Fix
req.body is undefined Missing middleware Add app.use(express.json()) BEFORE routes
Route never reached Wrong order Put specific routes before :id routes
Can't find expense params.id is string Use parseInt(req.params.id)
"Headers already sent" Sending two responses Add return after first response
